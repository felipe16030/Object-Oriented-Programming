Example Iterator/Iterable
import java.util.Iterator; // While Iterable is in java.lang, Iterator is in java.util
public class Alphabet implements Iterable<Character> { // Remember that generics must be Objects, not primitives
@Override
public Iterator<Character> iterator() { // Our Iterator must return the same type as our Iterable
return new AlphabetIterator();
}
private class AlphabetIterator implements Iterator<Character> { // Private inner class since nothing external needs to access the AlphabetIterator
private char c = 'a'; // You'll usually need some sort of variable to keep track of state (often an index)
@Override
public boolean hasNext() { // Determines whether or not there is more data
return c <= 'z';
}
@Override
public Character next() { // Gets the next piece of data
return c++;
}
}
public static void main(String[] args) {
Alphabet a = new Alphabet();
for(char c : a) { // Iterable allows us to use for-each, although Java hides the call to iterator()
System.out.print(c);
}
System.out.println();
}
}
